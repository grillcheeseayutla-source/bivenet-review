+import { supabase } from './supabase.mjs';
+import { rosCreateTempNat } from './routeros.mjs';
+import { executeSshCommand } from './ubiquiti.mjs';
+
+const FALLBACK_HARDCODED = [
+  { ssh_user: 'ubnt', ssh_pass: 'ubnt', label: 'Ubiquiti default' }
+];
+
+const TEST_COMMAND = 'echo ok';
+const TEST_TIMEOUT_MS = 8000;
+
+/**
+ * Prueba si una credencial SSH funciona contra un dispositivo.
+ * Usa el túnel NAT temporal del MikroTik.
+ * Retorna true si funciona, false si falla.
+ */
+async function testSSHCredential(api, routerVpnIp, deviceIp, user, pass) {
+  let natPort = null;
+  try {
+    const nat = await rosCreateTempNat(api, {
+      targetIp: deviceIp,
+      targetPort: 22
+    });
+    natPort = nat.port;
+
+    await new Promise(r => setTimeout(r, 1200));
+
+    // Race entre el comando SSH y un timeout
+    await Promise.race([
+      executeSshCommand(routerVpnIp, natPort, user, pass, TEST_COMMAND),
+      new Promise((_, reject) =>
+        setTimeout(() => reject(new Error('SSH test timeout')), TEST_TIMEOUT_MS)
+      )
+    ]);
+
+    return true;
+  } catch {
+    return false;
+  }
+}
+
+/**
+ * Guarda la credencial verificada en el dispositivo para
+ * evitar repetir el proceso la próxima vez.
+ */
+async function saveVerifiedCredentialToClient(deviceIp, wispId, user, pass) {
+  try {
+    await supabase
+      .from('isp_clients')
+      .update({
+        antenna_user: user,
+        antenna_pass: pass
+      })
+      .eq('ip_address', deviceIp)
+      .eq('wisp_id', wispId)
+      .is('antenna_user', null); // solo si no tiene una ya
+  } catch (e) {
+    console.warn('[CredResolver] No se pudo guardar credencial verificada:', e.message);
+  }
+}
+
+/**
+ * Incrementa contadores de éxito/fallo en el store.
+ */
+async function recordResult(credId, success) {
+  if (!credId) return;
+  try {
+    const field = success ? 'success_count' : 'fail_count';
+    await supabase.rpc('increment_ssh_cred_counter', {
+      cred_id: credId,
+      field_name: field
+    });
+  } catch {
+    // No crítico — no interrumpir el flujo
+  }
+}
+
+/**
+ * Resuelve las credenciales SSH para un dispositivo.
+ *
+ * Prioridad:
+ *   1. Credencial específica del dispositivo (isp_clients.antenna_user/pass)
+ *   2. Store del WISP ordenado por priority ASC
+ *   3. Fallback hardcodeado ubnt/ubnt
+ *
+ * Si encuentra la credencial correcta vía store,
+ * la guarda en isp_clients para próximas conexiones.
+ *
+ * @param {object} api - Instancia de RouterOSAPI ya conectada
+ * @param {string} routerVpnIp - IP VPN del MikroTik (para el túnel)
+ * @param {string} deviceIp - IP local del dispositivo Ubiquiti
+ * @param {string} wispId - UUID del WISP
+ * @param {string} deviceType - 'cpe' | 'ap' | 'ptp' | 'all'
+ * @returns {{ user: string, pass: string, source: string }}
+ */
+export async function resolveSSHCredentials(
+  api,
+  routerVpnIp,
+  deviceIp,
+  wispId,
+  deviceType = 'cpe'
+) {
+  console.log(`[CredResolver] Resolviendo credenciales para ${deviceIp} (type: ${deviceType})`);
+
+  // ── CAPA 1: Credencial específica del dispositivo ──────────
+  try {
+    const { data: client } = await supabase
+      .from('isp_clients')
+      .select('antenna_user, antenna_pass')
+      .eq('ip_address', deviceIp)
+      .eq('wisp_id', wispId)
+      .not('antenna_user', 'is', null)
+      .maybeSingle();
+
+    if (client?.antenna_user && client?.antenna_pass) {
+      console.log(`[CredResolver] ✅ Credencial específica encontrada en isp_clients`);
+      return {
+        user: client.antenna_user,
+        pass: client.antenna_pass,
+        source: 'client_specific'
+      };
+    }
+  } catch (e) {
+    console.warn('[CredResolver] Error buscando credencial específica:', e.message);
+  }
+
+  // ── CAPA 2: Store del WISP con fallback ordenado ───────────
+  let storeCredentials = [];
+  try {
+    const { data } = await supabase
+      .from('wisp_ssh_credentials')
+      .select('id, ssh_user, ssh_pass, label, priority')
+      .eq('wisp_id', wispId)
+      .in('device_type', [deviceType, 'all'])
+      .order('priority', { ascending: true });
+
+    storeCredentials = data || [];
+  } catch (e) {
+    console.warn('[CredResolver] Error cargando store de credenciales:', e.message);
+  }
+
+  // Combinar store + fallback hardcodeado al final
+  const allCredentials = [
+    ...storeCredentials,
+    ...FALLBACK_HARDCODED.map(c => ({ ...c, id: null }))
+  ];
+
+  console.log(`[CredResolver] Intentando ${allCredentials.length} credenciales para ${deviceIp}`);
+
+  for (const cred of allCredentials) {
+    console.log(`[CredResolver] → Probando: ${cred.ssh_user} / ${cred.label || 'fallback'}`);
+
+    const ok = await testSSHCredential(api, routerVpnIp, deviceIp, cred.ssh_user, cred.ssh_pass);
+
+    if (ok) {
+      console.log(`[CredResolver] ✅ Credencial válida: ${cred.ssh_user} (${cred.label || 'fallback'})`);
+      await recordResult(cred.id, true);
+
+      // Guardar en el cliente para próxima vez (solo si no tenía una)
+      await saveVerifiedCredentialToClient(deviceIp, wispId, cred.ssh_user, cred.ssh_pass);
+
+      return {
+        user: cred.ssh_user,
+        pass: cred.ssh_pass,
+        source: cred.id ? 'store_match' : 'hardcoded_fallback',
+        credId: cred.id,
+        label: cred.label
+      };
+    }
+
+    await recordResult(cred.id, false);
+  }
+
+  // ── Sin credenciales válidas ───────────────────────────────
+  throw new Error(
+    `[CredResolver] No se encontró credencial SSH válida para ${deviceIp}. ` +
+    `Se intentaron ${allCredentials.length} combinaciones. ` +
+    `Agrega las credenciales correctas en Configuración → Credenciales SSH.`
+  );
+}
