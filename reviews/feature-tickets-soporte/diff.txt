+'use server'
+
+import { createClient } from "@/lib/supabase/server"
+import { revalidatePath } from "next/cache"
+
+/**
+ * Obtiene los tickets de un cliente
+ */
+export async function getClientTicketsAction(
+  clientId: string,
+  wispId: string
+) {
+  try {
+    const supabase = await createClient()
+
+    // Verificar que el cliente pertenece al WISP
+    const { data: client, error: clientErr } = await supabase
+      .from('isp_clients')
+      .select('id')
+      .eq('id', clientId)
+      .eq('wisp_id', wispId)
+      .single()
+
+    if (clientErr || !client) {
+      throw new Error('Cliente no encontrado o sin permisos')
+    }
+
+    const { data, error } = await supabase
+      .from('support_tickets')
+      .select('id, title, description, status, priority, created_at, updated_at')
+      .eq('client_id', clientId)
+      .eq('wisp_id', wispId)
+      .order('created_at', { ascending: false })
+
+    if (error) throw error
+    return { success: true, data: data || [] }
+  } catch (error: any) {
+    return { success: false, error: error.message }
+  }
+}
+
+/**
+ * Crea un nuevo ticket de soporte para un cliente
+ */
+export async function createTicketAction(
+  clientId: string,
+  wispId: string,
+  ticket: {
+    title: string
+    description?: string
+    priority: 'low' | 'medium' | 'high'
+  }
+) {
+  try {
+    const supabase = await createClient()
+
+    // Verificar que el cliente pertenece al WISP
+    const { data: client, error: clientErr } = await supabase
+      .from('isp_clients')
+      .select('id')
+      .eq('id', clientId)
+      .eq('wisp_id', wispId)
+      .single()
+
+    if (clientErr || !client) {
+      throw new Error('Cliente no encontrado o sin permisos')
+    }
+
+    const { error } = await supabase
+      .from('support_tickets')
+      .insert({
+        wisp_id: wispId,
+        client_id: clientId,
+        title: ticket.title,
+        description: ticket.description || null,
+        priority: ticket.priority,
+        status: 'open'
+      })
+
+    if (error) throw error
+
+    revalidatePath('/dashboard/clientes')
+    return { success: true }
+  } catch (error: any) {
+    return { success: false, error: error.message }
+  }
+}
+
+/**
+ * Cambia el status de un ticket
+ */
+export async function updateTicketStatusAction(
+  ticketId: string,
+  wispId: string,
+  status: 'open' | 'in_progress' | 'closed'
+) {
+  try {
+    const supabase = await createClient()
+
+    const { error } = await supabase
+      .from('support_tickets')
+      .update({
+        status,
+        updated_at: new Date().toISOString()
+      })
+      .eq('id', ticketId)
+      .eq('wisp_id', wispId)
+
+    if (error) throw error
+
+    revalidatePath('/dashboard/clientes')
+    return { success: true }
+  } catch (error: any) {
+    return { success: false, error: error.message }
+  }
+}
