+'use server'
+
+import { createClient } from "@/lib/supabase/server"
+import { revalidatePath } from "next/cache"
+
+/**
+ * Obtiene la configuración del WISP actual
+ */
+export async function getWispSettingsAction() {
+  try {
+    const supabase = await createClient()
+
+    const { data: { user }, error: authErr } = await supabase.auth.getUser()
+    if (authErr || !user) throw new Error('No autenticado')
+
+    const { data: dbUser, error: userErr } = await supabase
+      .from('users')
+      .select('wisp_id')
+      .eq('id', user.id)
+      .single()
+
+    if (userErr || !dbUser?.wisp_id) throw new Error('WISP no encontrado')
+
+    const { data, error } = await supabase
+      .from('wisps')
+      .select(`
+        id,
+        nombre,
+        email_contacto,
+        telefono,
+        whatsapp,
+        logo_url,
+        ciudad,
+        timezone,
+        billing_day_default
+      `)
+      .eq('id', dbUser.wisp_id)
+      .single()
+
+    if (error) throw error
+    return { success: true, data }
+  } catch (error: any) {
+    return { success: false, error: error.message }
+  }
+}
+
+/**
+ * Actualiza la configuración del WISP actual
+ */
+export async function updateWispSettingsAction(settings: {
+  nombre?: string
+  email_contacto?: string
+  telefono?: string
+  whatsapp?: string
+  logo_url?: string
+  ciudad?: string
+  timezone?: string
+  billing_day_default?: number
+}) {
+  try {
+    const supabase = await createClient()
+
+    const { data: { user }, error: authErr } = await supabase.auth.getUser()
+    if (authErr || !user) throw new Error('No autenticado')
+
+    const { data: dbUser, error: userErr } = await supabase
+      .from('users')
+      .select('wisp_id')
+      .eq('id', user.id)
+      .single()
+
+    if (userErr || !dbUser?.wisp_id) throw new Error('WISP no encontrado')
+
+    const { error } = await supabase
+      .from('wisps')
+      .update({
+        ...settings,
+        updated_at: new Date().toISOString()
+      })
+      .eq('id', dbUser.wisp_id)
+
+    if (error) throw error
+
+    revalidatePath('/dashboard/configuracion')
+    return { success: true }
+  } catch (error: any) {
+    return { success: false, error: error.message }
+  }
+}
